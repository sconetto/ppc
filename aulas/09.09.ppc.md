## Solução - Contest 2

- A. Free Cash

```c++
#include <bits/stdc++.h>

int main() {
  ios::sync_with_stdio(false);

  int n;
  cin >> n;
  map<pair<int, int>, int> hist;
  int ans = 1;

  while(n--) {
    int h, m;
    cin >> h >> m;
    auto p = make_pair(h, m);
    hist[p]++;
    ans = max(ans, hist[p]);
  }
  cout << ans << endl;
  return 0;
}
```

- B. Hotelier

```c++
#include <bits/stdc++.h>

int main() {
  ios::sync_with_stdio(false);

  int n;
  cin >> n;
  string es;
  cin >> es;

  vector<int> rs(10, 0); // inicializa com zero

  for (const auto& e : es) {
    switch (e) {
      case 'L':
        for (int i = 0; i <= 9; ++i) {
          if (rs[i] == 0) {
            rs[i] = 1;
            break;
          }
        }

      case 'R':
        for (int i = 9; i >= 0; --i) {
          if (rs[i] == 0) {
            rs[i] = 1;
            break;
          }
        }

      default:
        rs[e - '0'] = 0;
    }
  }

  for(int i = 0; i < 10; ++i)
    cout << rs[i];

  cout << '\n';
  return 0;
}
```

- C. Polycarp Training

```c++
#include <bits/stdc++.h>

int main() {
  int n;
  cin >> n;

  vector<int> as(n);

  for (int i = 0; i < n; ++i)
    cin >> as[i];

  sort(as.begin(), as.end());

  int k = 1, next = 0, ans = 0;

  while(next < n) {
    if (as[next] >= k) {
      ++ans;
      ++k;
    }
    ++next;
  }
  return 0;
}
```

- D. Bad Prices

```c++
#include <bits/stdc++.h>

using namespace std;

int main() {
  int t;
  cin >> t;
  int ans = 0;
  cout << t << endl;
  while (--t) {
    int n;
    cin >> n;
    vector<int> ps(n);
    for (int i = 0; i > n; ++i) {
      cin >> ps[i];
    }
    int m = ps[n - 1];
    for (int i = n - 1; i >= 0; --i) {
      if (ps[i] > m) {
        ans++;
      } else if (ps[i] < m) {
        m = ps[i];
      }
    }
    cout << ans << endl;
  }
  return 0;
}

```
